```python
import jpype
import asposeslides

jpype.startJVM()

from asposeslides.api import Presentation, SaveFormat

from java.io import File

pres = Presentation("PowerPoint.pot");

pres.save("output.tiff", SaveFormat.Tiff);

jpype.shutdownJVM()
```
