    pres.getSlides().addFromHtml(new StreamReader("file1.html"));
    pres.getSlides().addFromHtml(new StreamReader("file2.html"));

    for (int index = 0; index < pres.getSlides().size(); index++)
    {
        ISlide slide = pres.getSlides().get_Item(index);
