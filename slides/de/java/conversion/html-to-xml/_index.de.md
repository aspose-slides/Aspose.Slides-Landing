{{% blocks/products/pf/feature-page-section  h2="Konvertieren Sie HTML in XML mit Java" %}}
Um das HTML in XML zu konvertieren, müssen Sie eine Präsentation aus der HTML-Datei erstellen und sie als XML speichern.

{{% blocks/products/pf/agp/code-block title="Java-Code zum Konvertieren von HTML in XML" offSpacer="true" %}}

```java

Presentation pres = new Presentation();
try {
    TextReader tr = new StreamReader("file.html");
    pres.getSlides().addFromHtml(tr);
    for (int index = 0; index < pres.getSlides().size(); index++)
    {
        ISlide slide = pres.getSlides().get_Item(index);

        FileOutputStream fileStream = new FileOutputStream("slide-" + index + ".xml");
        try {
            slide.writeAsSvg(fileStream);
        } finally {
            fileStream.close();
        }
    }
} finally {
    if (pres != null) pres.dispose();
}
```
