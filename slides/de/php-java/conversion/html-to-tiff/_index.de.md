{{% blocks/products/pf/feature-page-section  h2="Konvertieren Sie HTML in TIFF mit PHP" %}}
Um das HTML in TIFF zu konvertieren, müssen Sie eine Präsentation aus der HTML-Datei erstellen und sie als TIFF speichern.

{{% blocks/products/pf/agp/code-block title="PHP-Code zum Konvertieren von HTML in TIFF" offSpacer="true" %}}

```php

<?php
require_once("http://localhost:8080/JavaBridge/java/Java.inc");
require_once("lib/aspose.slides.php");
        
$pres = new Presentation();
try
{
    $pres->getSlides()->removeAt(0);
    
    $filename = 'file.html';
    $f = fopen($filename, 'r');
    if ($f) {
        $contents = fread($f, filesize($filename));
        fclose($f);
    }
    
    $pres->getSlides()->addFromHtml($contents);        
    $pres->save("output.tiff", SaveFormat::Tiff);        
}
finally
{
    if ($pres != null) $pres->dispose();
}
?>
```
