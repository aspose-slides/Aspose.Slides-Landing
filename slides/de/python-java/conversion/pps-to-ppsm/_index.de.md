```python

import jpype
import asposeslides

jpype.startJVM()

from asposeslides.api import Presentation, SaveFormat
from javax.imageio import ImageIO
from java.io import File

pres = Presentation("PowerPoint.pps");

pres.save("output.ppsm", SaveFormat.Ppsm);

jpype.shutdownJVM()
```
