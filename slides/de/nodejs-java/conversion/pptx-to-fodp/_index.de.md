Um das PPTX in FODP zu konvertieren, müssen Sie eine Präsentation aus der PPTX-Datei erstellen und diese als FODP speichern.

{{% blocks/products/pf/agp/code-block title="Node.js-Code zum Konvertieren von PPTX in FODP" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.pptx");
try
{
    pres.save("output.fodp", aspose.slides.SaveFormat.Fodp);
}
finally
{
    if (pres != null) pres.dispose();
}
```
