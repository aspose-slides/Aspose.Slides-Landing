{{% blocks/products/pf/feature-page-section  h2="Konvertieren Sie OTP in TIFF mit Node.js" %}}
Um das OTP in TIFF zu konvertieren, müssen Sie eine Präsentation aus der OTP-Datei erstellen und diese als TIFF speichern.

{{% blocks/products/pf/agp/code-block title="Node.js-Code zum Konvertieren von OTP in TIFF" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.otp");
try
{
    pres.save("output.tiff", aspose.slides.SaveFormat.Tiff);
}
finally
{
    if (pres != null) pres.dispose();
}
```
