Um das POTM in PPSX zu konvertieren, müssen Sie eine Präsentation aus der POTM-Datei erstellen und diese als PPSX speichern.

{{% blocks/products/pf/agp/code-block title="Node.js-Code zum Konvertieren von POTM in PPSX" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.potm");
try
{
    pres.save("output.ppsx", aspose.slides.SaveFormat.Ppsx);
}
finally
{
    if (pres != null) pres.dispose();
}
```
