Um das PPSX in ODP zu konvertieren, müssen Sie eine Präsentation aus der PPSX-Datei erstellen und diese als ODP speichern.

{{% blocks/products/pf/agp/code-block title="Node.js-Code zum Konvertieren von PPSX in ODP" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.ppsx");
try
{
    pres.save("output.odp", aspose.slides.SaveFormat.Odp);
}
finally
{
    if (pres != null) pres.dispose();
}
```
