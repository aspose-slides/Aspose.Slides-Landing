Um das POTX in GIF zu konvertieren, müssen Sie eine Präsentation aus der POTX-Datei erstellen und diese als GIF speichern.

{{% blocks/products/pf/agp/code-block title="Node.js-Code zum Konvertieren von POTX in GIF" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.potx");
try
{
    pres.save("output.gif", aspose.slides.SaveFormat.Gif);
}
finally
{
    if (pres != null) pres.dispose();
}
```
