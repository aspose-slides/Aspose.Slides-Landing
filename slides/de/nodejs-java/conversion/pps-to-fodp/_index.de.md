{{% blocks/products/pf/feature-page-section  h2="Konvertieren Sie PPS in FODP mit Node.js" %}}
Um das PPS in FODP zu konvertieren, müssen Sie eine Präsentation aus der PPS-Datei erstellen und diese als FODP speichern.

{{% blocks/products/pf/agp/code-block title="Node.js-Code zum Konvertieren von PPS in FODP" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.pps");
try
{
    pres.save("output.fodp", aspose.slides.SaveFormat.Fodp);
}
finally
{
    if (pres != null) pres.dispose();
}
```
