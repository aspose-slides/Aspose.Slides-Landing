{{% blocks/products/pf/feature-page-section  h2="Konvertieren Sie PDF in OTP mit Node.js" %}}
Um das PDF in OTP zu konvertieren, müssen Sie eine Präsentation aus der PDF-Datei erstellen und diese als OTP speichern.

{{% blocks/products/pf/agp/code-block title="Node.js-Code zum Konvertieren von PDF in OTP" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation();
try
{
    pres.getSlides().removeAt(0);
    pres.getSlides().addFromPdf("welcome-to-powerpoint.pdf");

    pres.save("output.otp", aspose.slides.SaveFormat.Otp);
}
finally
{
    if (pres != null) pres.dispose();
} 

```
