{{% blocks/products/pf/feature-page-section  h2="Konvertieren Sie ODP in POT mit Node.js" %}}
Um das ODP in POT zu konvertieren, müssen Sie eine Präsentation aus der ODP-Datei erstellen und diese als POT speichern.

{{% blocks/products/pf/agp/code-block title="Node.js-Code zum Konvertieren von ODP in POT" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.odp");
try
{
    pres.save("output.pot", aspose.slides.SaveFormat.Pot);
}
finally
{
    if (pres != null) pres.dispose();
}
```
