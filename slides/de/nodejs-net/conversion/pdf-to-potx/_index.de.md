Um das PDF in POTX zu konvertieren, müssen Sie eine Präsentation aus der PDF-Datei erstellen und diese als POTX speichern.

{{% blocks/products/pf/agp/code-block title="JavaScript-Code zum Konvertieren von PDF in POTX" offSpacer="true" %}}

```javascript

const fs = require('fs');
const asposeSlides = require('aspose.slides.via.net');
const { Presentation, SaveFormat, ShapeType } = asposeSlides;

var pres = new Presentation();
try
{
    pres.slides.removeAt(0);
    var slides = pres.slides.addFromPdf("welcome-to-powerpoint.pdf");

    pres.save("output.potx", SaveFormat.Potx);
}
finally
{
    if (pres != null) pres.dispose();
} 

```
