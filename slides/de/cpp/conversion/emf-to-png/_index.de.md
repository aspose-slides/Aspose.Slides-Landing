// Load the EMF

// User defined dimension
int desiredX = 1200;
int desiredY = 800;

// Getting scaled value  of X and Y
float ScaleX = (float)(1.0 / pres->get_SlideSize()->get_Size().get_Width()) * desiredX;

// Create a custom scale image
auto bitmap = slide->GetThumbnail(ScaleX, ScaleY);
