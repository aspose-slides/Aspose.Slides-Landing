{{% blocks/products/pf/agp/code-block title="Kod Node.js do konwersji PPSM na POTM" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.ppsm");
try
{
    pres.save("output.potm", aspose.slides.SaveFormat.Potm);
}
finally
{
    if (pres != null) pres.dispose();
}
```
