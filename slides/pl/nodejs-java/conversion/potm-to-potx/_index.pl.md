{{% blocks/products/pf/agp/code-block title="Kod Node.js do konwersji POTM na POTX" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.potm");
try
{
    pres.save("output.potx", aspose.slides.SaveFormat.Potx);
}
finally
{
    if (pres != null) pres.dispose();
}
```
