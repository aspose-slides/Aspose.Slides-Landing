```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.fodp");
try
{
    pres.save("output.potx", aspose.slides.SaveFormat.Potx);
}
finally
{
    if (pres != null) pres.dispose();
}
```
