Aby przekonwertować plik PDF na GIF, musisz utworzyć prezentację z pliku PDF i zapisać ją jako GIF.

{{% blocks/products/pf/agp/code-block title="Kod Node.js do konwersji PDF na GIF" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation();
try
{
    pres.getSlides().removeAt(0);
    pres.getSlides().addFromPdf("welcome-to-powerpoint.pdf");

    pres.save("output.gif", aspose.slides.SaveFormat.Gif);
}
finally
{
    if (pres != null) pres.dispose();
} 

```
