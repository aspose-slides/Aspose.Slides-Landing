{{% blocks/products/pf/agp/code-block title="Kod Node.js do konwersji POT na HTML" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.pot");
try
{
    pres.save("output.html", aspose.slides.SaveFormat.Html5);
}
finally
{
    if (pres != null) pres.dispose();
}
```
