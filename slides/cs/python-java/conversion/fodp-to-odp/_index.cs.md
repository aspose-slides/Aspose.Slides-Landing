```python

import jpype
import asposeslides

jpype.startJVM()

from asposeslides.api import Presentation, SaveFormat
from javax.imageio import ImageIO
from java.io import File

pres = Presentation("PowerPoint.fodp");

pres.save("output.odp", SaveFormat.Odp);

jpype.shutdownJVM()
```
