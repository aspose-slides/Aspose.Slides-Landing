var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.otp");
try
{
    pres.save("output.fodp", aspose.slides.SaveFormat.Fodp);
}
finally
{
    if (pres != null) pres.dispose();
}
