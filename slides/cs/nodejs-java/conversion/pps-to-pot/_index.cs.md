{{% blocks/products/pf/agp/code-block title="Kód Node.js pro převod PPS do POT" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.pps");
try
{
    pres.save("output.pot", aspose.slides.SaveFormat.Pot);
}
finally
{
    if (pres != null) pres.dispose();
}
```
