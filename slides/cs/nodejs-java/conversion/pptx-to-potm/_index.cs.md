var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.pptx");
try
{
    pres.save("output.potm", aspose.slides.SaveFormat.Potm);
}
finally
{
    if (pres != null) pres.dispose();
}
