{{% blocks/products/pf/agp/code-block title="Kód Node.js pro převod OTP do ODP" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.otp");
try
{
    pres.save("output.odp", aspose.slides.SaveFormat.Odp);
}
finally
{
    if (pres != null) pres.dispose();
}
```
