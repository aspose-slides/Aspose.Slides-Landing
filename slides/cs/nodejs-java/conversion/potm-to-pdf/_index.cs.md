{{% blocks/products/pf/agp/code-block title="Kód Node.js pro převod POTM do PDF" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.potm");
try
{
    pres.save("output.pdf", aspose.slides.SaveFormat.Pdf);
}
finally
{
    if (pres != null) pres.dispose();
}
```
