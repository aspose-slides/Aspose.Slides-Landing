{{% blocks/products/pf/agp/code-block title="Kód Node.js pro převod ODP do GIF" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.odp");
try
{
    pres.save("output.gif", aspose.slides.SaveFormat.Gif);
}
finally
{
    if (pres != null) pres.dispose();
}
```
