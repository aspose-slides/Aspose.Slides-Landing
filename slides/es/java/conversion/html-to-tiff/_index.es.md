```java

Presentation pres = new Presentation();
try {
    TextReader tr = new StreamReader("file.html");
    pres.getSlides().addFromHtml(tr);
    pres.save("doc.tiff", SaveFormat.Tiff);
} finally {
    if (pres != null) pres.dispose();
}
```
