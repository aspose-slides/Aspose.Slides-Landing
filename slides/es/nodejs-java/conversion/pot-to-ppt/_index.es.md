{{% blocks/products/pf/agp/code-block title="Código Node.js para convertir POT en PPT" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.pot");
try
{
    pres.save("output.ppt", aspose.slides.SaveFormat.Ppt);
}
finally
{
    if (pres != null) pres.dispose();
}
```
