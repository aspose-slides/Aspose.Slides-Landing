using (Presentation pres = new Presentation())

        IPPImage image = pres.Images.AddImage(File.ReadAllBytes("image.png"));

        pres.Slides[0].GetThumbnail(new Size(960, 720)).Save($"image.jpg", ImageFormat.Jpeg);
    }
