import jpype
import asposeslides

jpype.startJVM()

from asposeslides.api import Presentation, SaveFormat
from javax.imageio import ImageIO

pres = Presentation();

pres.getSlides().removeAt(0);

pres.save("output.ppt", SaveFormat.Ppt);
