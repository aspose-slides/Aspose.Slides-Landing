{{% blocks/products/pf/agp/code-block title="Tutoriel Python pour convertir POT en POTM" offSpacer="true" %}}

```python

import jpype
import asposeslides

jpype.startJVM()

from asposeslides.api import Presentation, SaveFormat
from javax.imageio import ImageIO
from java.io import File

pres = Presentation("PowerPoint.pot");

pres.save("output.potm", SaveFormat.Potm);

jpype.shutdownJVM()
```
