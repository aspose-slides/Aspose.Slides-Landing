{{% blocks/products/pf/agp/code-block title="Python oktatóanyag a POT HTML formátumba való konvertálásához" offSpacer="true" %}}

```python

import jpype
import asposeslides

jpype.startJVM()

from asposeslides.api import Presentation, SaveFormat
from javax.imageio import ImageIO
from java.io import File

pres = Presentation("PowerPoint.pot");

pres.save("pres.html", SaveFormat.Html);

jpype.shutdownJVM()
```
