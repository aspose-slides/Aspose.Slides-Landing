{{% blocks/products/pf/agp/code-block title="Node.js kód a POTM FODP formátumba való konvertálásához" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.potm");
try
{
    pres.save("output.fodp", aspose.slides.SaveFormat.Fodp);
}
finally
{
    if (pres != null) pres.dispose();
}
```
