```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.pptm");
try
{
    pres.save("output.pptx", aspose.slides.SaveFormat.Pptx);
}
finally
{
    if (pres != null) pres.dispose();
}
```
