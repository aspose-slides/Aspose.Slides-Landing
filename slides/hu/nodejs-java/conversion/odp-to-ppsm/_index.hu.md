{{% blocks/products/pf/agp/code-block title="Node.js kód a ODP PPSM formátumba való konvertálásához" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.odp");
try
{
    pres.save("output.ppsm", aspose.slides.SaveFormat.Ppsm);
}
finally
{
    if (pres != null) pres.dispose();
}
```
