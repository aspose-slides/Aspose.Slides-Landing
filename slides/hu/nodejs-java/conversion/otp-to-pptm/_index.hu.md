{{% blocks/products/pf/agp/code-block title="Node.js kód a OTP PPTM formátumba való konvertálásához" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.otp");
try
{
    pres.save("output.pptm", aspose.slides.SaveFormat.Pptm);
}
finally
{
    if (pres != null) pres.dispose();
}
```
