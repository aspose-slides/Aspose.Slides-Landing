Presentation pres = new Presentation();
try {
    pres.getSlides().addFromPdf("InputPDF1.pdf");
    pres.getSlides().addFromPdf("InputPDF2.pdf");

    pres.save("pres.pdf", SaveFormat.Pdf);
} finally {
    if (pres != null) pres.dispose();
}
