{{% blocks/products/pf/agp/code-block title="Kode Node.js untuk mengubah POTX menjadi PPS" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.potx");
try
{
    pres.save("output.pps", aspose.slides.SaveFormat.Pps);
}
finally
{
    if (pres != null) pres.dispose();
}
```
