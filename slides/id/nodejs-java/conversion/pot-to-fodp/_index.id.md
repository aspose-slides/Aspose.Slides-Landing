{{% blocks/products/pf/agp/code-block title="Kode Node.js untuk mengubah POT menjadi FODP" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.pot");
try
{
    pres.save("output.fodp", aspose.slides.SaveFormat.Fodp);
}
finally
{
    if (pres != null) pres.dispose();
}
```
