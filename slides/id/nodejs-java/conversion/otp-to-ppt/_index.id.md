{{% blocks/products/pf/agp/code-block title="Kode Node.js untuk mengubah OTP menjadi PPT" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.otp");
try
{
    pres.save("output.ppt", aspose.slides.SaveFormat.Ppt);
}
finally
{
    if (pres != null) pres.dispose();
}
```
