{{% blocks/products/pf/agp/code-block title="Kode Node.js untuk mengubah ODP menjadi PPTM" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.odp");
try
{
    pres.save("output.pptm", aspose.slides.SaveFormat.Pptm);
}
finally
{
    if (pres != null) pres.dispose();
}
```
