```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.potx");
try
{
    pres.save("output.gif", aspose.slides.SaveFormat.Gif);
}
finally
{
    if (pres != null) pres.dispose();
}
```
