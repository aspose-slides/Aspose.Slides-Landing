{{% blocks/products/pf/agp/code-block title="Kode Node.js untuk mengubah PPT menjadi OTP" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.ppt");
try
{
    pres.save("output.otp", aspose.slides.SaveFormat.Otp);
}
finally
{
    if (pres != null) pres.dispose();
}
```
