Untuk mengonversi PPT ke SVG, Anda perlu membuat Presentasi dari file PPT dan menyimpannya sebagai SVG.

{{% blocks/products/pf/agp/code-block title="Kode Node.js untuk mengubah PPT menjadi SVG" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.ppt");
try
{
    var slide = pres.getSlides().get_Item(0);
    var svgStream = java.newInstanceSync("java.io.FileOutputStream", "image.svg");
    slide.writeAsSvg(svgStream);
    svgStream.close();
}
finally
{
    if (pres != null) pres.dispose();
}
```
