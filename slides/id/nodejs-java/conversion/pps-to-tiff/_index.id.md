{{% blocks/products/pf/agp/code-block title="Kode Node.js untuk mengubah PPS menjadi TIFF" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.pps");
try
{
    pres.save("output.tiff", aspose.slides.SaveFormat.Tiff);
}
finally
{
    if (pres != null) pres.dispose();
}
```
