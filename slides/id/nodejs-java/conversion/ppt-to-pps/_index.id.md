{{% blocks/products/pf/agp/code-block title="Kode Node.js untuk mengubah PPT menjadi PPS" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.ppt");
try
{
    pres.save("output.pps", aspose.slides.SaveFormat.Pps);
}
finally
{
    if (pres != null) pres.dispose();
}
```
