{{% blocks/products/pf/agp/code-block title="Tutorial Python para converter POT em SVG" offSpacer="true" %}}

```python

import jpype
import asposeslides

jpype.startJVM()

from asposeslides.api import Presentation, SaveFormat
from javax.imageio import ImageIO
from java.io import File

pres = Presentation("PowerPoint.pot");

for i in range(pres.getSlides().size()):
    outputStream = open('slide" + i + ".svg', "wb")
    outputStream.write(Slide.writeAsSvgToBytes(pres.getSlides().get_Item(i)))

jpype.shutdownJVM()
```
