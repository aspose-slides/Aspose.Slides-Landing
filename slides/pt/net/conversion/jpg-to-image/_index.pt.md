using (Presentation pres = new Presentation())

        IPPImage image = pres.Images.AddImage(File.ReadAllBytes("image.jpg"));

        pres.Slides[0].GetThumbnail(new Size(960, 720)).Save($"image.png", ImageFormat.Png);
    }
