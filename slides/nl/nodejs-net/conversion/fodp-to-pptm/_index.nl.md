{{% blocks/products/pf/agp/code-block title="JavaScript-code voor het converteren van FODP naar PPTM" offSpacer="true" %}}

```javascript

const fs = require('fs');
const asposeSlides = require('aspose.slides.via.net');
const { Presentation, SaveFormat } = asposeSlides;
var pres = new Presentation("welcome-to-powerpoint.fodp");
try
{
    pres.save("output.pptm", SaveFormat.Pptm);
}
finally
{
    if (pres != null) pres.dispose();
}
```
