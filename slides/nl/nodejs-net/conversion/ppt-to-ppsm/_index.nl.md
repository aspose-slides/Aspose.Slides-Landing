{{% blocks/products/pf/agp/code-block title="JavaScript-code voor het converteren van PPT naar PPSM" offSpacer="true" %}}

```javascript

const fs = require('fs');
const asposeSlides = require('aspose.slides.via.net');
const { Presentation, SaveFormat } = asposeSlides;
var pres = new Presentation("welcome-to-powerpoint.ppt");
try
{
    pres.save("output.ppsm", SaveFormat.Ppsm);
}
finally
{
    if (pres != null) pres.dispose();
}
```
