{{% blocks/products/pf/agp/code-block title="JavaScript-code voor het converteren van PPS naar POTX" offSpacer="true" %}}

```javascript

const fs = require('fs');
const asposeSlides = require('aspose.slides.via.net');
const { Presentation, SaveFormat } = asposeSlides;
var pres = new Presentation("welcome-to-powerpoint.pps");
try
{
    pres.save("output.potx", SaveFormat.Potx);
}
finally
{
    if (pres != null) pres.dispose();
}
```
