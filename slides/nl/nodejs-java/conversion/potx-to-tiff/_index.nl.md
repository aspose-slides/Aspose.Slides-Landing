Om het bestand POTX naar TIFF te converteren, moet u een presentatie maken van het bestand POTX en dit opslaan als TIFF.

{{% blocks/products/pf/agp/code-block title="Node.js-code voor het converteren van POTX naar TIFF" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.potx");
try
{
    pres.save("output.tiff", aspose.slides.SaveFormat.Tiff);
}
finally
{
    if (pres != null) pres.dispose();
}
```
