Om het bestand OTP naar PPTX te converteren, moet u een presentatie maken van het bestand OTP en dit opslaan als PPTX.

{{% blocks/products/pf/agp/code-block title="Node.js-code voor het converteren van OTP naar PPTX" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.otp");
try
{
    pres.save("output.pptx", aspose.slides.SaveFormat.Pptx);
}
finally
{
    if (pres != null) pres.dispose();
}
```
