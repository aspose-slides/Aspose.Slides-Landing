Om het bestand PDF naar PPSX te converteren, moet u een presentatie maken van het bestand PDF en dit opslaan als PPSX.

{{% blocks/products/pf/agp/code-block title="Node.js-code voor het converteren van PDF naar PPSX" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation();
try
{
    pres.getSlides().removeAt(0);
    pres.getSlides().addFromPdf("welcome-to-powerpoint.pdf");

    pres.save("output.ppsx", aspose.slides.SaveFormat.Ppsx);
}
finally
{
    if (pres != null) pres.dispose();
} 

```
