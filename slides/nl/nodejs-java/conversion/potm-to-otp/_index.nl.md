Om het bestand POTM naar OTP te converteren, moet u een presentatie maken van het bestand POTM en dit opslaan als OTP.

{{% blocks/products/pf/agp/code-block title="Node.js-code voor het converteren van POTM naar OTP" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.potm");
try
{
    pres.save("output.otp", aspose.slides.SaveFormat.Otp);
}
finally
{
    if (pres != null) pres.dispose();
}
```
