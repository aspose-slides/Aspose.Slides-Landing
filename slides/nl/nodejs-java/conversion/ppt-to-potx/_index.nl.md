Om het bestand PPT naar POTX te converteren, moet u een presentatie maken van het bestand PPT en dit opslaan als POTX.

{{% blocks/products/pf/agp/code-block title="Node.js-code voor het converteren van PPT naar POTX" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.ppt");
try
{
    pres.save("output.potx", aspose.slides.SaveFormat.Potx);
}
finally
{
    if (pres != null) pres.dispose();
}
```
