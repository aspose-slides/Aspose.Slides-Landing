Om het bestand PPTX naar ODP te converteren, moet u een presentatie maken van het bestand PPTX en dit opslaan als ODP.

{{% blocks/products/pf/agp/code-block title="Node.js-code voor het converteren van PPTX naar ODP" offSpacer="true" %}}

```javascript

var aspose = aspose || {};

aspose.slides = require("aspose.slides.via.java");

var pres = new aspose.slides.Presentation("welcome-to-powerpoint.pptx");
try
{
    pres.save("output.odp", aspose.slides.SaveFormat.Odp);
}
finally
{
    if (pres != null) pres.dispose();
}
```
